📌 STATUS CODES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

P    = Present (Regular working day)
A    = Absent (Not marked present)
WO   = Weekly Off (Scheduled day off)
WOP  = Weekly Off - Present (Worked on day off)
H    = Holiday (Company/National Holiday)
HP   = Holiday Present (Worked on holiday)
PL   = Paid Leave / Personal Leave
C OFF = Compensatory Off

⏱️ TIME FORMAT
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

USE 24-HOUR FORMAT:

✓ CORRECT:     ✗ WRONG:
  09:00          9:00 AM
  09:30          09:30 a.m.
  17:45          5:45 PM
  18:00          6 PM

OFFICE HOURS (Default):
  In Time:  09:00
  Out Time: 18:00

⚡ LATE ARRIVAL CATEGORIES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

A1  = ≤ 10 minutes late
A2  = > 10 and ≤ 30 minutes late
A3  = > 30 and ≤ 45 minutes late
A4  = > 45 minutes late

Example: Employee arrives at 09:20 (20 min late) = Category A2

⏭️ EARLY DEPARTURE CATEGORIES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

B1  = ≤ 10 minutes early
B2  = > 10 and ≤ 30 minutes early
B3  = > 30 and ≤ 45 minutes early
B4  = > 45 minutes early

Example: Employee leaves at 17:30 (30 min early) = Category B2

🎮 HOW TO USE FILTERS & SORT
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

FILTER BY STATUS:
┌────────────────────────────────────────┐
│ All Status  - Show all records         │
│ P           - Show only Present days   │
│ A           - Show only Absent days    │
│ WO          - Show Weekly Off days     │
│ H           - Show Holiday records     │
│ PL          - Show Leave records       │
│ HP          - Show Holiday Present     │
└────────────────────────────────────────┘

SORT BY:
  • Date       - Chronological order
  • Status     - Group by attendance status
  • Late Time  - Sort by arrival time
  • Early Time - Sort by departure time

ORDER:
  ↑ Ascending  - A to Z, 1 to 9
  ↓ Descending - Z to A, 9 to 1

🛠️ TROUBLESHOOTING QUICK FIXES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

ISSUE                          → SOLUTION
─────────────────────────────────────────────────────────────────────────
File size exceeds 10MB         → Reduce file size / Split data
Employee name not found        → Add "Employee:" label in first 20 rows
Day headers not recognized     → Format: "1 W", "2 Th" (number + day)
Status row not found           → Ensure row labeled "Status" exists
Invalid time format            → Use HH:MM (24-hour): 09:30, not 9:30 AM
Wrong totals                   → Check status codes (P, A, WO, etc.)
Browser not working            → Try Chrome, Firefox, Safari, or Edge
Slow processing                → Check internet connection
Can't open file                → Ensure .xls or .xlsx format

❓ FREQUENTLY ASKED QUESTIONS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Q: Is my data saved?
A: No! Data is processed in your browser and NOT saved on any server.

Q: Can I process multiple employees?
A: Yes, upload each employee's file separately.

Q: What if I have data for only 20 days?
A: That's fine! The app adapts to any number of days in your file.

Q: Can I change office hours?
A: This requires code modification. Default is 9:00 AM to 6:00 PM.

Q: Does it work on mobile?
A: Yes, the app is fully mobile-responsive!

Q: Can I filter by date range?
A: Edit your Excel file to include only desired dates before uploading.

Q: What browsers work best?
A: Chrome, Firefox, Safari, Edge (latest versions recommended).

Q: Is internet required?
A: Yes, active internet connection is needed.
